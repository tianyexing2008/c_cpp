275 int
276 _dbus_read_socket_with_unix_fds (int fd,
277  DBusString *buffer,
278  int count,
279  int *fds,
280  int *n_fds) {
281 #ifndef HAVE_UNIX_FD_PASSING
282  int r;
283 
284  if ((r = _dbus_read_socket(fd, buffer, count)) < 0)
285  return r;
286 
287  *n_fds = 0;
288  return r;
289 
290 #else
291  int bytes_read;
292  int start;
293  struct msghdr m;
294  struct iovec iov;
295 
296  _dbus_assert (count >= 0);
297  _dbus_assert (*n_fds >= 0);
298 
299  start = _dbus_string_get_length (buffer);
300 
301  if (!_dbus_string_lengthen (buffer, count))
302  {
303  errno = ENOMEM;
304  return -1;
305  }
306 
307  _DBUS_ZERO(iov);
308  iov.iov_base = _dbus_string_get_data_len (buffer, start, count);
309  iov.iov_len = count;
310 
311  _DBUS_ZERO(m);
312  m.msg_iov = &iov;
313  m.msg_iovlen = 1;
314 
315  /* Hmm, we have no clue how long the control data will actually be
316  that is queued for us. The least we can do is assume that the
317  caller knows. Hence let's make space for the number of fds that
318  we shall read at max plus the cmsg header. */
319  m.msg_controllen = CMSG_SPACE(*n_fds * sizeof(int));
320 
321  /* It's probably safe to assume that systems with SCM_RIGHTS also
322  know alloca() */
323  m.msg_control = alloca(m.msg_controllen);
324  memset(m.msg_control, 0, m.msg_controllen);
325 
326  /* Do not include the padding at the end when we tell the kernel
327  * how much we're willing to receive. This avoids getting
328  * the padding filled with additional fds that we weren't expecting,
329  * if a (potentially malicious) sender included them. (fd.o #83622) */
330  m.msg_controllen = CMSG_LEN (*n_fds * sizeof(int));
331 
332  again:
333 
334  bytes_read = recvmsg(fd, &m, 0
335 #ifdef MSG_CMSG_CLOEXEC
336  |MSG_CMSG_CLOEXEC
337 #endif
338  );
339 
340  if (bytes_read < 0)
341  {
342  if (errno == EINTR)
343  goto again;
344  else
345  {
346  /* put length back (note that this doesn't actually realloc anything) */
347  _dbus_string_set_length (buffer, start);
348  return -1;
349  }
350  }
351  else
352  {
353  struct cmsghdr *cm;
354  dbus_bool_t found = FALSE;
355 
356  if (m.msg_flags & MSG_CTRUNC)
357  {
358  /* Hmm, apparently the control data was truncated. The bad
359  thing is that we might have completely lost a couple of fds
360  without chance to recover them. Hence let's treat this as a
361  serious error. */
362 
363  errno = ENOSPC;
364  _dbus_string_set_length (buffer, start);
365  return -1;
366  }
367 
368  for (cm = CMSG_FIRSTHDR(&m); cm; cm = CMSG_NXTHDR(&m, cm))
369  if (cm->cmsg_level == SOL_SOCKET && cm->cmsg_type == SCM_RIGHTS)
370  {
371  size_t i;
372  int *payload = (int *) CMSG_DATA (cm);
373  size_t payload_len_bytes = (cm->cmsg_len - CMSG_LEN (0));
374  size_t payload_len_fds = payload_len_bytes / sizeof (int);
375  size_t fds_to_use;
376 
377  /* Every non-negative int fits in a size_t without truncation,
378  * and we already know that *n_fds is non-negative, so
379  * casting (size_t) *n_fds is OK */
380  _DBUS_STATIC_ASSERT (sizeof (size_t) >= sizeof (int));
381 
382  if (_DBUS_LIKELY (payload_len_fds <= (size_t) *n_fds))
383  {
384  /* The fds in the payload will fit in our buffer */
385  fds_to_use = payload_len_fds;
386  }
387  else
388  {
389  /* Too many fds in the payload. This shouldn't happen
390  * any more because we're setting m.msg_controllen to
391  * the exact number we can accept, but be safe and
392  * truncate. */
393  fds_to_use = (size_t) *n_fds;
394 
395  /* Close the excess fds to avoid DoS: if they stayed open,
396  * someone could send us an extra fd per message
397  * and we'd eventually run out. */
398  for (i = fds_to_use; i < payload_len_fds; i++)
399  {
400  close (payload[i]);
401  }
402  }
403 
404  memcpy (fds, payload, fds_to_use * sizeof (int));
405  found = TRUE;
406  /* This cannot overflow because we have chosen fds_to_use
407  * to be <= *n_fds */
408  *n_fds = (int) fds_to_use;
409 
410  /* Linux doesn't tell us whether MSG_CMSG_CLOEXEC actually
411  worked, hence we need to go through this list and set
412  CLOEXEC everywhere in any case */
413  for (i = 0; i < fds_to_use; i++)
414  _dbus_fd_set_close_on_exec(fds[i]);
415 
416  break;
417  }
418 
419  if (!found)
420  *n_fds = 0;
421 
422  /* put length back (doesn't actually realloc) */
423  _dbus_string_set_length (buffer, start + bytes_read);
424 
425 #if 0
426  if (bytes_read > 0)
427  _dbus_verbose_bytes_of_string (buffer, start, bytes_read);
428 #endif
429 
430  return bytes_read;
431  }
432 #endif
433 }

832 int
833 _dbus_connect_unix_socket (const char *path,
834  dbus_bool_t abstract,
835  DBusError *error)
836 {
837  int fd;
838  size_t path_len;
839  struct sockaddr_un addr;
840 
841  _DBUS_ASSERT_ERROR_IS_CLEAR (error);
842 
843  _dbus_verbose ("connecting to unix socket %s abstract=%d\n",
844  path, abstract);
845 
846 
847  if (!_dbus_open_unix_socket (&fd, error))
848  {
849  _DBUS_ASSERT_ERROR_IS_SET(error);
850  return -1;
851  }
852  _DBUS_ASSERT_ERROR_IS_CLEAR(error);
853 
854  _DBUS_ZERO (addr);
855  addr.sun_family = AF_UNIX;
856  path_len = strlen (path);
857 
858  if (abstract)
859  {
860 #ifdef HAVE_ABSTRACT_SOCKETS
861  addr.sun_path[0] = '\0'; /* this is what says "use abstract" */
862  path_len++; /* Account for the extra nul byte added to the start of sun_path */
863 
864  if (path_len > _DBUS_MAX_SUN_PATH_LENGTH)
865  {
866  dbus_set_error (error, DBUS_ERROR_BAD_ADDRESS,
867  "Abstract socket name too long\n");
868  _dbus_close (fd, NULL);
869  return -1;
870  }
871 
872  strncpy (&addr.sun_path[1], path, path_len);
873  /* _dbus_verbose_bytes (addr.sun_path, sizeof (addr.sun_path)); */
874 #else /* HAVE_ABSTRACT_SOCKETS */
875  dbus_set_error (error, DBUS_ERROR_NOT_SUPPORTED,
876  "Operating system does not support abstract socket namespace\n");
877  _dbus_close (fd, NULL);
878  return -1;
879 #endif /* ! HAVE_ABSTRACT_SOCKETS */
880  }
