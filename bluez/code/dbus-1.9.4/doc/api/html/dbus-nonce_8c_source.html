33 static dbus_bool_t
34 do_check_nonce (int fd, const DBusString *nonce, DBusError *error)
35 {
36  DBusString buffer;
37  DBusString p;
38  size_t nleft;
39  dbus_bool_t result;
40  int n;
41 
42  _DBUS_ASSERT_ERROR_IS_CLEAR (error);
43 
44  nleft = 16;
45 
46  if ( !_dbus_string_init (&buffer)
47  || !_dbus_string_init (&p) ) {
48  dbus_set_error (error, DBUS_ERROR_NO_MEMORY, NULL);
49  _dbus_string_free (&p);
50  _dbus_string_free (&buffer);
51  return FALSE;
52  }
53 
54  while (nleft)
55  {
56  int saved_errno;
57 
58  n = _dbus_read_socket (fd, &p, nleft);
59  saved_errno = _dbus_save_socket_errno ();
60 
61  if (n == -1 && _dbus_get_is_errno_eintr (saved_errno))
62  ;
63  else if (n == -1 && _dbus_get_is_errno_eagain_or_ewouldblock (saved_errno))
64  _dbus_sleep_milliseconds (100);
65  else if (n==-1)
66  {
67  dbus_set_error (error, DBUS_ERROR_IO_ERROR, "Could not read nonce from socket (fd=%d)", fd );
68  _dbus_string_free (&p);
69  _dbus_string_free (&buffer);
70  return FALSE;
71  }
72  else if (!n)
73  {
74  _dbus_string_free (&p);
75  _dbus_string_free (&buffer);
76  dbus_set_error (error, DBUS_ERROR_IO_ERROR, "Could not read nonce from socket (fd=%d)", fd );
77  return FALSE;
78  }
79  else
80  {
81  _dbus_string_append_len(&buffer, _dbus_string_get_const_data (&p), n);
82  nleft -= n;
83  }
84  }
85 
86  result = _dbus_string_equal_len (&buffer, nonce, 16);
87  if (!result)
88  dbus_set_error (error, DBUS_ERROR_ACCESS_DENIED, "Nonces do not match, access denied (fd=%d)", fd );
89 
90  _dbus_string_free (&p);
91  _dbus_string_free (&buffer);
92 
93  return result;
94 }

104 dbus_bool_t
105 _dbus_read_nonce (const DBusString *fname, DBusString *nonce, DBusError* error)
106 {
107  FILE *fp;
108  char buffer[17];
109  size_t nread;
110 
111  buffer[sizeof buffer - 1] = '\0';
112 
113  _DBUS_ASSERT_ERROR_IS_CLEAR (error);
114 
115  _dbus_verbose ("reading nonce from file: %s\n", _dbus_string_get_const_data (fname));
116 
117 
118  fp = fopen (_dbus_string_get_const_data (fname), "rb");
119  if (!fp)
120  {
121  dbus_set_error (error,
122  _dbus_error_from_system_errno (),
123  "Failed to open %s for read: %s",
124  _dbus_string_get_const_data (fname),
125  _dbus_strerror_from_errno ());
126  return FALSE;
127  }
128 
129  nread = fread (buffer, 1, sizeof buffer - 1, fp);
130  fclose (fp);
131  if (!nread)
132  {
133  dbus_set_error (error, DBUS_ERROR_FILE_NOT_FOUND, "Could not read nonce from file %s", _dbus_string_get_const_data (fname));
134  return FALSE;
135  }
136 
137  if (!_dbus_string_append_len (nonce, buffer, sizeof buffer - 1 ))
138  {
139  dbus_set_error (error, DBUS_ERROR_NO_MEMORY, NULL);
140  return FALSE;
141  }
142  return TRUE;
143 }

145 int
146 _dbus_accept_with_noncefile (int listen_fd, const DBusNonceFile *noncefile)
147 {
148  int fd;
149  DBusString nonce;
150 
151  _dbus_assert (noncefile != NULL);
152  if (!_dbus_string_init (&nonce))
153  return -1;
154  //PENDING(kdab): set better errors
155  if (_dbus_read_nonce (_dbus_noncefile_get_path(noncefile), &nonce, NULL) != TRUE)
156  return -1;
157  fd = _dbus_accept (listen_fd);
158  if (_dbus_socket_is_invalid (fd))
159  return fd;
160  if (do_check_nonce(fd, &nonce, NULL) != TRUE) {
161  _dbus_verbose ("nonce check failed. Closing socket.\n");
162  _dbus_close_socket(fd, NULL);
163  return -1;
164  }
165 
166  return fd;
167 }

169 static dbus_bool_t
170 generate_and_write_nonce (const DBusString *filename, DBusError *error)
171 {
172  DBusString nonce;
173  dbus_bool_t ret;
174 
175  _DBUS_ASSERT_ERROR_IS_CLEAR (error);
176 
177  if (!_dbus_string_init (&nonce))
178  {
179  dbus_set_error (error, DBUS_ERROR_NO_MEMORY, NULL);
180  return FALSE;
181  }
182 
183  if (!_dbus_generate_random_bytes (&nonce, 16))
184  {

186  _dbus_string_free (&nonce);

189 
190  ret = _dbus_string_save_to_file (&nonce, filename, FALSE, error);
191 
192  _dbus_string_free (&nonce);

206 dbus_bool_t
207 _dbus_send_nonce (int fd, const DBusString *noncefile, DBusError *error)
208 {
209  dbus_bool_t read_result;
210  int send_result;
211  DBusString nonce;
212 
213  _DBUS_ASSERT_ERROR_IS_CLEAR (error);
214 
215  if (_dbus_string_get_length (noncefile) == 0)
216  return FALSE;
217 
218  if (!_dbus_string_init (&nonce))
219  {
220  dbus_set_error (error, DBUS_ERROR_NO_MEMORY, NULL);
221  return FALSE;
222  }
223 
224  read_result = _dbus_read_nonce (noncefile, &nonce, error);
225  if (!read_result)
226  {
227  _DBUS_ASSERT_ERROR_IS_SET (error);
228  _dbus_string_free (&nonce);
229  return FALSE;
230  }
231  _DBUS_ASSERT_ERROR_IS_CLEAR (error);
232 
233  send_result = _dbus_write_socket (fd, &nonce, 0, _dbus_string_get_length (&nonce));
234 
235  _dbus_string_free (&nonce);
236 
237  if (send_result == -1)
238  {
239  dbus_set_error (error,
240  _dbus_error_from_system_errno (),
241  "Failed to send nonce (fd=%d): %s",
242  fd, _dbus_strerror_from_errno ());
243  return FALSE;
244  }
245 
246  return TRUE;
247 }

249 static dbus_bool_t
250 do_noncefile_create (DBusNonceFile *noncefile,
251  DBusError *error,
252  dbus_bool_t use_subdir)
253 {
254  DBusString randomStr;
255  const char *tmp;
256 
257  _DBUS_ASSERT_ERROR_IS_CLEAR (error);
258 
259  _dbus_assert (noncefile);
260 
261  if (!_dbus_string_init (&randomStr))
262  {
263  dbus_set_error (error, DBUS_ERROR_NO_MEMORY, NULL);
264  goto on_error;
265  }
266 
267  if (!_dbus_generate_random_ascii (&randomStr, 8))
268  {
269  dbus_set_error (error, DBUS_ERROR_NO_MEMORY, NULL);
270  goto on_error;
271  }
272 
273  tmp = _dbus_get_tmpdir ();
274 
275  if (!_dbus_string_init (&noncefile->dir)
276  || tmp == NULL
277  || !_dbus_string_append (&noncefile->dir, tmp))
278  {
279  dbus_set_error (error, DBUS_ERROR_NO_MEMORY, NULL);
280  goto on_error;
281  }
282  if (use_subdir)
283  {
284  if (!_dbus_string_append (&noncefile->dir, "/dbus_nonce-")
285  || !_dbus_string_append (&noncefile->dir, _dbus_string_get_const_data (&randomStr)) )
286  {
287  dbus_set_error (error, DBUS_ERROR_NO_MEMORY, NULL);
288  goto on_error;
289  }
290  if (!_dbus_string_init (&noncefile->path)
291  || !_dbus_string_copy (&noncefile->dir, 0, &noncefile->path, 0)
292  || !_dbus_string_append (&noncefile->path, "/nonce"))
293  {
294  dbus_set_error (error, DBUS_ERROR_NO_MEMORY, NULL);
295  goto on_error;
296  }
297  if (!_dbus_create_directory (&noncefile->dir, error))
298  {
299  _DBUS_ASSERT_ERROR_IS_SET (error);
300  goto on_error;
301  }
302  _DBUS_ASSERT_ERROR_IS_CLEAR (error);
303 
304  }
305  else
306  {
307  if (!_dbus_string_init (&noncefile->path)
308  || !_dbus_string_copy (&noncefile->dir, 0, &noncefile->path, 0)
309  || !_dbus_string_append (&noncefile->path, "/dbus_nonce-")
310  || !_dbus_string_append (&noncefile->path, _dbus_string_get_const_data (&randomStr)))
311  {
312  dbus_set_error (error, DBUS_ERROR_NO_MEMORY, NULL);
313  goto on_error;
314  }
315 
316  }
317 
318  if (!generate_and_write_nonce (&noncefile->path, error))
319  {
320  _DBUS_ASSERT_ERROR_IS_SET (error);
321  if (use_subdir)
322  _dbus_delete_directory (&noncefile->dir, NULL); //we ignore possible errors deleting the dir and return the write error instead
323  goto on_error;
324  }
325  _DBUS_ASSERT_ERROR_IS_CLEAR (error);
326 
327  _dbus_string_free (&randomStr);
328 
329  return TRUE;
330  on_error:
331  if (use_subdir)
332  _dbus_delete_directory (&noncefile->dir, NULL);
333  _dbus_string_free (&noncefile->dir);
334  _dbus_string_free (&noncefile->path);
335  _dbus_string_free (&randomStr);
336  return FALSE;
337 }
