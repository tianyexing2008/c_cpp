61 socket_finalize (DBusServer *server)
62 {
63  DBusServerSocket *socket_server = (DBusServerSocket*) server;
64  int i;
65 
66  _dbus_server_finalize_base (server);
67 
68  for (i = 0 ; i < socket_server->n_fds ; i++)
69  if (socket_server->watch[i])
70  {
71  _dbus_watch_unref (socket_server->watch[i]);
72  socket_server->watch[i] = NULL;
73  }
74 
75  dbus_free (socket_server->fds);
76  dbus_free (socket_server->watch);
77  dbus_free (socket_server->socket_name);

80  dbus_free (socket_server->noncefile);
81  dbus_free (server);
82 }

85 static dbus_bool_t
86 handle_new_client_fd_and_unlock (DBusServer *server,
87  int client_fd)
88 {
89  DBusConnection *connection;
90  DBusTransport *transport;
91  DBusNewConnectionFunction new_connection_function;
92  void *new_connection_data;
93 
94  _dbus_verbose ("Creating new client connection with fd %d\n", client_fd);
95 
96  HAVE_LOCK_CHECK (server);
97 
98  if (!_dbus_set_fd_nonblocking (client_fd, NULL))
99  {
100  SERVER_UNLOCK (server);
101  return TRUE;
102  }
103 
104  transport = _dbus_transport_new_for_socket (client_fd, &server->guid_hex, NULL);
105  if (transport == NULL)
106  {
107  _dbus_close_socket (client_fd, NULL);
108  SERVER_UNLOCK (server);
109  return FALSE;
110  }
111 
112  if (!_dbus_transport_set_auth_mechanisms (transport,
113  (const char **) server->auth_mechanisms))
114  {
115  _dbus_transport_unref (transport);
116  SERVER_UNLOCK (server);
117  return FALSE;
118  }
119 
120  /* note that client_fd is now owned by the transport, and will be
121  * closed on transport disconnection/finalization
122  */
123 
124  connection = _dbus_connection_new_for_transport (transport);
125  _dbus_transport_unref (transport);
126  transport = NULL; /* now under the connection lock */
127 
128  if (connection == NULL)
129  {
130  SERVER_UNLOCK (server);
131  return FALSE;
132  }
133 
134  /* See if someone wants to handle this new connection, self-referencing
135  * for paranoia.
136  */
137  new_connection_function = server->new_connection_function;
138  new_connection_data = server->new_connection_data;
139 
140  _dbus_server_ref_unlocked (server);
141  SERVER_UNLOCK (server);
142 
143  if (new_connection_function)
144  {
145  (* new_connection_function) (server, connection,
146  new_connection_data);
147  }
148  dbus_server_unref (server);
149 
150  /* If no one grabbed a reference, the connection will die. */
151  _dbus_connection_close_if_only_one_ref (connection);
152  dbus_connection_unref (connection);
153 
154  return TRUE;
155 }

157 static dbus_bool_t
158 socket_handle_watch (DBusWatch *watch,
159  unsigned int flags,
160  void *data)
161 {
162  DBusServer *server = data;
163  DBusServerSocket *socket_server = data;
164 
165 #ifndef DBUS_DISABLE_ASSERT
166  int i;
167  dbus_bool_t found = FALSE;
168 #endif
169 
170  SERVER_LOCK (server);
171 
172 #ifndef DBUS_DISABLE_ASSERT
173  for (i = 0 ; i < socket_server->n_fds ; i++)
174  {
175  if (socket_server->watch[i] == watch)
176  found = TRUE;
177  }
178  _dbus_assert (found);
179 #endif
180 
181  _dbus_verbose ("Handling client connection, flags 0x%x\n", flags);
182 
183  if (flags & DBUS_WATCH_READABLE)
184  {
185  int client_fd;
186  int listen_fd;
187  int saved_errno;
188 
189  listen_fd = dbus_watch_get_socket (watch);
190 
191  if (socket_server->noncefile)
192  client_fd = _dbus_accept_with_noncefile (listen_fd, socket_server->noncefile);
193  else
194  client_fd = _dbus_accept (listen_fd);
195 
196  saved_errno = _dbus_save_socket_errno ();
197 
198  if (client_fd < 0)
199  {
200  /* EINTR handled for us */
201 
202  if (_dbus_get_is_errno_eagain_or_ewouldblock (saved_errno))
203  _dbus_verbose ("No client available to accept after all\n");
204  else
205  _dbus_verbose ("Failed to accept a client connection: %s\n",
206  _dbus_strerror (saved_errno));
207 
208  SERVER_UNLOCK (server);
209  }
210  else
211  {
212  if (!handle_new_client_fd_and_unlock (server, client_fd))
213  _dbus_verbose ("Rejected client connection due to lack of memory\n");
214  }
215  }
216 
217  if (flags & DBUS_WATCH_ERROR)
218  _dbus_verbose ("Error on server listening socket\n");
219 
220  if (flags & DBUS_WATCH_HANGUP)
221  _dbus_verbose ("Hangup on server listening socket\n");
222 
223  return TRUE;
224 }

226 static void
227 socket_disconnect (DBusServer *server)
228 {
229  DBusServerSocket *socket_server = (DBusServerSocket*) server;
230  int i;
231 
232  HAVE_LOCK_CHECK (server);
233 
234  for (i = 0 ; i < socket_server->n_fds ; i++)
235  {
236  if (socket_server->watch[i])
237  {
238  _dbus_server_remove_watch (server,
239  socket_server->watch[i]);
240  _dbus_watch_invalidate (socket_server->watch[i]);
241  _dbus_watch_unref (socket_server->watch[i]);
242  socket_server->watch[i] = NULL;
243  }
244 
245  _dbus_close_socket (socket_server->fds[i], NULL);
246  socket_server->fds[i] = -1;
247  }
248 
249  if (socket_server->socket_name != NULL)
250  {
251  DBusString tmp;
252  _dbus_string_init_const (&tmp, socket_server->socket_name);
253  _dbus_delete_file (&tmp, NULL);
254  }
255 
256  if (server->published_address)
257  _dbus_daemon_unpublish_session_bus_address();
258 
259  HAVE_LOCK_CHECK (server);
260 }

282 DBusServer*
283 _dbus_server_new_for_socket (int *fds,
284  int n_fds,
285  const DBusString *address,
286  DBusNonceFile *noncefile)
287 {
288  DBusServerSocket *socket_server;
289  DBusServer *server;
290  int i;
291 
292  socket_server = dbus_new0 (DBusServerSocket, 1);
293  if (socket_server == NULL)
294  return NULL;
295 
296  socket_server->noncefile = noncefile;
297 
298  socket_server->fds = dbus_new (int, n_fds);
299  if (!socket_server->fds)
300  goto failed_0;
301 
302  socket_server->watch = dbus_new0 (DBusWatch *, n_fds);
303  if (!socket_server->watch)
304  goto failed_1;
305 
306  for (i = 0 ; i < n_fds ; i++)
307  {
308  DBusWatch *watch;
309 
310  watch = _dbus_watch_new (fds[i],
311  DBUS_WATCH_READABLE,
312  TRUE,
313  socket_handle_watch, socket_server,
314  NULL);
315  if (watch == NULL)
316  goto failed_2;
317 
318  socket_server->n_fds++;
319  socket_server->fds[i] = fds[i];
320  socket_server->watch[i] = watch;
321  }
322 
323  if (!_dbus_server_init_base (&socket_server->base,
324  &socket_vtable, address))
325  goto failed_2;
326 
327  server = (DBusServer*)socket_server;
328 
329  SERVER_LOCK (server);
330 
331  for (i = 0 ; i < n_fds ; i++)
332  {
333  if (!_dbus_server_add_watch (&socket_server->base,
334  socket_server->watch[i]))
335  {
336  int j;
337  for (j = 0 ; j < i ; j++)
338  _dbus_server_remove_watch (server,
339  socket_server->watch[j]);
340 
341  SERVER_UNLOCK (server);
342  _dbus_server_finalize_base (&socket_server->base);
343  goto failed_2;
344  }
345  }
346 
347  SERVER_UNLOCK (server);
348 
349  _dbus_server_trace_ref (&socket_server->base, 0, 1, "new_for_socket");
350  return (DBusServer*) socket_server;
351 
352  failed_2:
353  for (i = 0 ; i < n_fds ; i++)
354  {
355  if (socket_server->watch[i] != NULL)
356  {
357  _dbus_watch_unref (socket_server->watch[i]);
358  socket_server->watch[i] = NULL;
359  }
360  }
361  dbus_free (socket_server->watch);
362 
363  failed_1:
364  dbus_free (socket_server->fds);
365 
366  failed_0:
367  dbus_free (socket_server);
368  return NULL;
369 }

390 DBusServer*
391 _dbus_server_new_for_tcp_socket (const char *host,
392  const char *bind,
393  const char *port,
394  const char *family,
395  DBusError *error,
396  dbus_bool_t use_nonce)
397 {
398  DBusServer *server;
399  int *listen_fds = NULL;
400  int nlisten_fds = 0, i;
401  DBusString address;
402  DBusString host_str;
403  DBusString port_str;
404  DBusNonceFile *noncefile;
405 
406  _DBUS_ASSERT_ERROR_IS_CLEAR (error);
407 
408  noncefile = NULL;
409 
410  if (!_dbus_string_init (&address))
411  {
412  dbus_set_error (error, DBUS_ERROR_NO_MEMORY, NULL);
413  return NULL;
414  }
415 
416  if (!_dbus_string_init (&port_str))
417  {
418  dbus_set_error (error, DBUS_ERROR_NO_MEMORY, NULL);
419  goto failed_0;
420  }
421 
422  if (host == NULL)
423  host = "localhost";
424 
425  if (port == NULL)
426  port = "0";
427 
428  if (bind == NULL)
429  bind = host;
430  else if (strcmp (bind, "*") == 0)
431  bind = NULL;
432 
433  nlisten_fds =_dbus_listen_tcp_socket (bind, port, family,
434  &port_str,
435  &listen_fds, error);
436  if (nlisten_fds <= 0)
437  {
438  _DBUS_ASSERT_ERROR_IS_SET(error);
439  goto failed_1;
440  }
441 
442  _dbus_string_init_const (&host_str, host);
443  if (!_dbus_string_append (&address, use_nonce ? "nonce-tcp:host=" : "tcp:host=") ||
444  !_dbus_address_append_escaped (&address, &host_str) ||
445  !_dbus_string_append (&address, ",port=") ||
446  !_dbus_string_append (&address, _dbus_string_get_const_data(&port_str)))
447  {
448  dbus_set_error (error, DBUS_ERROR_NO_MEMORY, NULL);
449  goto failed_2;
450  }
451  if (family &&
452  (!_dbus_string_append (&address, ",family=") ||
453  !_dbus_string_append (&address, family)))
454  {
455  dbus_set_error (error, DBUS_ERROR_NO_MEMORY, NULL);
456  goto failed_2;
457  }
458 
459  if (use_nonce)
460  {
461  noncefile = dbus_new0 (DBusNonceFile, 1);
462  if (noncefile == NULL)
463  {
464  dbus_set_error (error, DBUS_ERROR_NO_MEMORY, NULL);
465  goto failed_2;
466  }
467 
468  if (!_dbus_noncefile_create (noncefile, error))
469  goto failed_3;
470 
471  if (!_dbus_string_append (&address, ",noncefile=") ||
472  !_dbus_address_append_escaped (&address, _dbus_noncefile_get_path (noncefile)))
473  {
474  dbus_set_error (error, DBUS_ERROR_NO_MEMORY, NULL);
475  goto failed_4;
476  }
477 
478  }
479 
480  server = _dbus_server_new_for_socket (listen_fds, nlisten_fds, &address, noncefile);
481  if (server == NULL)
482  {
483  dbus_set_error (error, DBUS_ERROR_NO_MEMORY, NULL);
484  if (noncefile != NULL)
485  goto failed_4;
486  else
487  goto failed_2;
488  }
489 
490  _dbus_string_free (&port_str);
491  _dbus_string_free (&address);
492  dbus_free(listen_fds);
493 
494  return server;
495 
496  failed_4:
497  _dbus_noncefile_delete (noncefile, NULL);
498 
499  failed_3:
500  dbus_free (noncefile);
501 
502  failed_2:
503  for (i = 0 ; i < nlisten_fds ; i++)
504  _dbus_close_socket (listen_fds[i], NULL);
505  dbus_free(listen_fds);
506 
507  failed_1:
508  _dbus_string_free (&port_str);
509 
510  failed_0:
511  _dbus_string_free (&address);
512 
513  return NULL;
514 }

528 DBusServerListenResult
529 _dbus_server_listen_socket (DBusAddressEntry *entry,
530  DBusServer **server_p,
531  DBusError *error)
532 {
533  const char *method;
534 
535  *server_p = NULL;
536 
537  method = dbus_address_entry_get_method (entry);
538 
539  if (strcmp (method, "tcp") == 0 || strcmp (method, "nonce-tcp") == 0)
540  {
541  const char *host;
542  const char *port;
543  const char *bind;
544  const char *family;
545 
546  host = dbus_address_entry_get_value (entry, "host");
547  bind = dbus_address_entry_get_value (entry, "bind");
548  port = dbus_address_entry_get_value (entry, "port");
549  family = dbus_address_entry_get_value (entry, "family");
550 
551  *server_p = _dbus_server_new_for_tcp_socket (host, bind, port,
552  family, error, strcmp (method, "nonce-tcp") == 0 ? TRUE : FALSE);
553 
554  if (*server_p)
555  {
556  _DBUS_ASSERT_ERROR_IS_CLEAR(error);
557  return DBUS_SERVER_LISTEN_OK;
558  }
559  else
560  {
561  _DBUS_ASSERT_ERROR_IS_SET(error);
562  return DBUS_SERVER_LISTEN_DID_NOT_CONNECT;
563  }
564  }
565  else
566  {
567  _DBUS_ASSERT_ERROR_IS_CLEAR(error);
568  return DBUS_SERVER_LISTEN_NOT_HANDLED;
569  }
570 }
