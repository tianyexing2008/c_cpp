121 static void
122 check_write_watch (DBusTransport *transport)
123 {
124  DBusTransportSocket *socket_transport = (DBusTransportSocket*) transport;
125  dbus_bool_t needed;
126 
127  if (transport->connection == NULL)
128  return;
129 
130  if (transport->disconnected)
131  {
132  _dbus_assert (socket_transport->write_watch == NULL);
133  return;
134  }
135 
136  _dbus_transport_ref (transport);
137 
138  if (_dbus_transport_try_to_authenticate (transport))
139  needed = _dbus_connection_has_messages_to_send_unlocked (transport->connection);
140  else
141  {
142  if (transport->send_credentials_pending)
143  needed = TRUE;
144  else
145  {
146  DBusAuthState auth_state;
147 
148  auth_state = _dbus_auth_do_work (transport->auth);
149 
150  /* If we need memory we install the write watch just in case,
151  * if there's no need for it, it will get de-installed
152  * next time we try reading.
153  */
154  if (auth_state == DBUS_AUTH_STATE_HAVE_BYTES_TO_SEND ||
155  auth_state == DBUS_AUTH_STATE_WAITING_FOR_MEMORY)
156  needed = TRUE;
157  else
158  needed = FALSE;
159  }
160  }
161 
162  _dbus_verbose ("check_write_watch(): needed = %d on connection %p watch %p fd = %d outgoing messages exist %d\n",
163  needed, transport->connection, socket_transport->write_watch,
164  socket_transport->fd,
165  _dbus_connection_has_messages_to_send_unlocked (transport->connection));
166 
167  _dbus_connection_toggle_watch_unlocked (transport->connection,
168  socket_transport->write_watch,
169  needed);
170 
171  _dbus_transport_unref (transport);
172 }

174 static void
175 check_read_watch (DBusTransport *transport)
176 {
177  DBusTransportSocket *socket_transport = (DBusTransportSocket*) transport;
178  dbus_bool_t need_read_watch;
179 
180  _dbus_verbose ("fd = %d\n",socket_transport->fd);
181 
182  if (transport->connection == NULL)
183  return;
184 
185  if (transport->disconnected)
186  {
187  _dbus_assert (socket_transport->read_watch == NULL);
188  return;
189  }
190 
191  _dbus_transport_ref (transport);
192 
193  if (_dbus_transport_try_to_authenticate (transport))
194  need_read_watch =
195  (_dbus_counter_get_size_value (transport->live_messages) < transport->max_live_messages_size) &&
196  (_dbus_counter_get_unix_fd_value (transport->live_messages) < transport->max_live_messages_unix_fds);
197  else
198  {
199  if (transport->receive_credentials_pending)
200  need_read_watch = TRUE;
201  else
202  {
203  /* The reason to disable need_read_watch when not WAITING_FOR_INPUT
204  * is to avoid spinning on the file descriptor when we're waiting
205  * to write or for some other part of the auth process
206  */
207  DBusAuthState auth_state;
208 
209  auth_state = _dbus_auth_do_work (transport->auth);
210 
211  /* If we need memory we install the read watch just in case,
212  * if there's no need for it, it will get de-installed
213  * next time we try reading. If we're authenticated we
214  * install it since we normally have it installed while
215  * authenticated.
216  */
217  if (auth_state == DBUS_AUTH_STATE_WAITING_FOR_INPUT ||
218  auth_state == DBUS_AUTH_STATE_WAITING_FOR_MEMORY ||
219  auth_state == DBUS_AUTH_STATE_AUTHENTICATED)
220  need_read_watch = TRUE;
221  else
222  need_read_watch = FALSE;
223  }
224  }
225 
226  _dbus_verbose (" setting read watch enabled = %d\n", need_read_watch);
227  _dbus_connection_toggle_watch_unlocked (transport->connection,
228  socket_transport->read_watch,
229  need_read_watch);
230 
231  _dbus_transport_unref (transport);
232 }

242 /* return value is whether we successfully read any new data. */
243 static dbus_bool_t
244 read_data_into_auth (DBusTransport *transport,
245  dbus_bool_t *oom)
246 {
247  DBusTransportSocket *socket_transport = (DBusTransportSocket*) transport;
248  DBusString *buffer;
249  int bytes_read;
250  int saved_errno;
251 
252  *oom = FALSE;
253 
254  _dbus_auth_get_buffer (transport->auth, &buffer);
255 
256  bytes_read = _dbus_read_socket (socket_transport->fd,
257  buffer, socket_transport->max_bytes_read_per_iteration);
258  saved_errno = _dbus_save_socket_errno ();
259 
260  _dbus_auth_return_buffer (transport->auth, buffer);
261 
262  if (bytes_read > 0)
263  {
264  _dbus_verbose (" read %d bytes in auth phase\n", bytes_read);
265 
266  return TRUE;
267  }
268  else if (bytes_read < 0)
269  {
270  /* EINTR already handled for us */
271 
272  if (_dbus_get_is_errno_enomem (saved_errno))
273  {
274  *oom = TRUE;
275  }
276  else if (_dbus_get_is_errno_eagain_or_ewouldblock (saved_errno))
277  ; /* do nothing, just return FALSE below */
278  else
279  {
280  _dbus_verbose ("Error reading from remote app: %s\n",
281  _dbus_strerror (saved_errno));
282  do_io_error (transport);
283  }
284 
285  return FALSE;
286  }
287  else
288  {
289  _dbus_assert (bytes_read == 0);
290 
291  _dbus_verbose ("Disconnected from remote app\n");
292  do_io_error (transport);
293 
294  return FALSE;
295  }
296 }

299 static dbus_bool_t
300 write_data_from_auth (DBusTransport *transport)
301 {
302  DBusTransportSocket *socket_transport = (DBusTransportSocket*) transport;
303  int bytes_written;
304  int saved_errno;
305  const DBusString *buffer;
306 
307  if (!_dbus_auth_get_bytes_to_send (transport->auth,
308  &buffer))
309  return FALSE;
310 
311  bytes_written = _dbus_write_socket (socket_transport->fd,
312  buffer,
313  0, _dbus_string_get_length (buffer));
314  saved_errno = _dbus_save_socket_errno ();
315 
316  if (bytes_written > 0)
317  {
318  _dbus_auth_bytes_sent (transport->auth, bytes_written);
319  return TRUE;
320  }
321  else if (bytes_written < 0)
322  {
323  /* EINTR already handled for us */
324 
325  if (_dbus_get_is_errno_eagain_or_ewouldblock (saved_errno))
326  ;
327  else
328  {
329  _dbus_verbose ("Error writing to remote app: %s\n",
330  _dbus_strerror (saved_errno));
331  do_io_error (transport);
332  }
333  }
334 
335  return FALSE;
336 }

338 /* FALSE on OOM */
339 static dbus_bool_t
340 exchange_credentials (DBusTransport *transport,
341  dbus_bool_t do_reading,
342  dbus_bool_t do_writing)
343 {
344  DBusTransportSocket *socket_transport = (DBusTransportSocket*) transport;
345  DBusError error = DBUS_ERROR_INIT;
346 
347  _dbus_verbose ("exchange_credentials: do_reading = %d, do_writing = %d\n",
348  do_reading, do_writing);
349 
350  if (do_writing && transport->send_credentials_pending)
351  {
352  if (_dbus_send_credentials_socket (socket_transport->fd,
353  &error))
354  {
355  transport->send_credentials_pending = FALSE;
356  }
357  else
358  {
359  _dbus_verbose ("Failed to write credentials: %s\n", error.message);
360  dbus_error_free (&error);
361  do_io_error (transport);
362  }
363  }
364 
365  if (do_reading && transport->receive_credentials_pending)
366  {
367  /* FIXME this can fail due to IO error _or_ OOM, broken
368  * (somewhat tricky to fix since the OOM error can be set after
369  * we already read the credentials byte, so basically we need to
370  * separate reading the byte and storing it in the
371  * transport->credentials). Does not really matter for now
372  * because storing in credentials never actually fails on unix.
373  */
374  if (_dbus_read_credentials_socket (socket_transport->fd,
375  transport->credentials,
376  &error))
377  {
378  transport->receive_credentials_pending = FALSE;
379  }
380  else
381  {
382  _dbus_verbose ("Failed to read credentials %s\n", error.message);
383  dbus_error_free (&error);
384  do_io_error (transport);
385  }
386  }
387 
388  if (!(transport->send_credentials_pending ||
389  transport->receive_credentials_pending))
390  {
391  if (!_dbus_auth_set_credentials (transport->auth,
392  transport->credentials))
393  return FALSE;
394  }
395 
396  return TRUE;
397 }

399 static dbus_bool_t
400 do_authentication (DBusTransport *transport,
401  dbus_bool_t do_reading,
402  dbus_bool_t do_writing,
403  dbus_bool_t *auth_completed)
404 {
405  dbus_bool_t oom;
406  dbus_bool_t orig_auth_state;
407 
408  oom = FALSE;
409 
410  orig_auth_state = _dbus_transport_try_to_authenticate (transport);
411 
412  /* This is essential to avoid the check_write_watch() at the end,
413  * we don't want to add a write watch in do_iteration before
414  * we try writing and get EAGAIN
415  */
416  if (orig_auth_state)
417  {
418  if (auth_completed)
419  *auth_completed = FALSE;
420  return TRUE;
421  }
422 
423  _dbus_transport_ref (transport);
424 
425  while (!_dbus_transport_try_to_authenticate (transport) &&
426  _dbus_transport_get_is_connected (transport))
427  {
428  if (!exchange_credentials (transport, do_reading, do_writing))
429  {
430  /* OOM */
431  oom = TRUE;
432  goto out;
433  }
434 
435  if (transport->send_credentials_pending ||
436  transport->receive_credentials_pending)
437  {
438  _dbus_verbose ("send_credentials_pending = %d receive_credentials_pending = %d\n",
439  transport->send_credentials_pending,
440  transport->receive_credentials_pending);
441  goto out;
442  }
443 
444 #define TRANSPORT_SIDE(t) ((t)->is_server ? "server" : "client")
445  switch (_dbus_auth_do_work (transport->auth))
446  {
447  case DBUS_AUTH_STATE_WAITING_FOR_INPUT:
448  _dbus_verbose (" %s auth state: waiting for input\n",
449  TRANSPORT_SIDE (transport));
450  if (!do_reading || !read_data_into_auth (transport, &oom))
451  goto out;
452  break;
453 
454  case DBUS_AUTH_STATE_WAITING_FOR_MEMORY:
455  _dbus_verbose (" %s auth state: waiting for memory\n",
456  TRANSPORT_SIDE (transport));
457  oom = TRUE;
458  goto out;
459  break;
460 
461  case DBUS_AUTH_STATE_HAVE_BYTES_TO_SEND:
462  _dbus_verbose (" %s auth state: bytes to send\n",
463  TRANSPORT_SIDE (transport));
464  if (!do_writing || !write_data_from_auth (transport))
465  goto out;
466  break;
467 
468  case DBUS_AUTH_STATE_NEED_DISCONNECT:
469  _dbus_verbose (" %s auth state: need to disconnect\n",
470  TRANSPORT_SIDE (transport));
471  do_io_error (transport);
472  break;
473 
474  case DBUS_AUTH_STATE_AUTHENTICATED:
475  _dbus_verbose (" %s auth state: authenticated\n",
476  TRANSPORT_SIDE (transport));
477  break;
478  }
479  }
480 
481  out:
482  if (auth_completed)
483  *auth_completed = (orig_auth_state != _dbus_transport_try_to_authenticate (transport));
484 
485  check_read_watch (transport);
486  check_write_watch (transport);
487  _dbus_transport_unref (transport);
488 
489  if (oom)
490  return FALSE;
491  else
492  return TRUE;
493 }

495 /* returns false on oom */
496 static dbus_bool_t
497 do_writing (DBusTransport *transport)
498 {
499  int total;
500  DBusTransportSocket *socket_transport = (DBusTransportSocket*) transport;
501  dbus_bool_t oom;
502 
503  /* No messages without authentication! */
504  if (!_dbus_transport_try_to_authenticate (transport))
505  {
506  _dbus_verbose ("Not authenticated, not writing anything\n");
507  return TRUE;
508  }
509 
510  if (transport->disconnected)
511  {
512  _dbus_verbose ("Not connected, not writing anything\n");
513  return TRUE;
514  }
515 
516 #if 1
517  _dbus_verbose ("do_writing(), have_messages = %d, fd = %d\n",
518  _dbus_connection_has_messages_to_send_unlocked (transport->connection),
519  socket_transport->fd);
520 #endif
521 
522  oom = FALSE;
523  total = 0;
524 
525  while (!transport->disconnected &&
526  _dbus_connection_has_messages_to_send_unlocked (transport->connection))
527  {
528  int bytes_written;
529  DBusMessage *message;
530  const DBusString *header;
531  const DBusString *body;
532  int header_len, body_len;
533  int total_bytes_to_write;
534  int saved_errno;
535 
536  if (total > socket_transport->max_bytes_written_per_iteration)
537  {
538  _dbus_verbose ("%d bytes exceeds %d bytes written per iteration, returning\n",
539  total, socket_transport->max_bytes_written_per_iteration);
540  goto out;
541  }
542 
543  message = _dbus_connection_get_message_to_send (transport->connection);
544  _dbus_assert (message != NULL);
545  dbus_message_lock (message);
546 
547 #if 0
548  _dbus_verbose ("writing message %p\n", message);
549 #endif
550 
551  _dbus_message_get_network_data (message,
552  &header, &body);
553 
554  header_len = _dbus_string_get_length (header);
555  body_len = _dbus_string_get_length (body);
556 
557  if (_dbus_auth_needs_encoding (transport->auth))
558  {
559  /* Does fd passing even make sense with encoded data? */
560  _dbus_assert(!DBUS_TRANSPORT_CAN_SEND_UNIX_FD(transport));
561 
562  if (_dbus_string_get_length (&socket_transport->encoded_outgoing) == 0)
563  {
564  if (!_dbus_auth_encode_data (transport->auth,
565  header, &socket_transport->encoded_outgoing))
566  {
567  oom = TRUE;
568  goto out;
569  }
570 
571  if (!_dbus_auth_encode_data (transport->auth,
572  body, &socket_transport->encoded_outgoing))
573  {
574  _dbus_string_set_length (&socket_transport->encoded_outgoing, 0);
575  oom = TRUE;
576  goto out;
577  }
578  }
579 
580  total_bytes_to_write = _dbus_string_get_length (&socket_transport->encoded_outgoing);
581 
582 #if 0
583  _dbus_verbose ("encoded message is %d bytes\n",
584  total_bytes_to_write);
585 #endif
586 
587  bytes_written =
588  _dbus_write_socket (socket_transport->fd,
589  &socket_transport->encoded_outgoing,
590  socket_transport->message_bytes_written,
591  total_bytes_to_write - socket_transport->message_bytes_written);
592  saved_errno = _dbus_save_socket_errno ();
593  }
594  else
595  {
596  total_bytes_to_write = header_len + body_len;
597 
598 #if 0
599  _dbus_verbose ("message is %d bytes\n",
600  total_bytes_to_write);
601 #endif
602 
603 #ifdef HAVE_UNIX_FD_PASSING
604  if (socket_transport->message_bytes_written <= 0 && DBUS_TRANSPORT_CAN_SEND_UNIX_FD(transport))
605  {
606  /* Send the fds along with the first byte of the message */
607  const int *unix_fds;
608  unsigned n;
609 
610  _dbus_message_get_unix_fds(message, &unix_fds, &n);
611 
612  bytes_written =
613  _dbus_write_socket_with_unix_fds_two (socket_transport->fd,
614  header,
615  socket_transport->message_bytes_written,
616  header_len - socket_transport->message_bytes_written,
617  body,
618  0, body_len,
619  unix_fds,
620  n);
621  saved_errno = _dbus_save_socket_errno ();
622 
623  if (bytes_written > 0 && n > 0)
624  _dbus_verbose("Wrote %i unix fds\n", n);
625  }
626  else
627 #endif
628  {
629  if (socket_transport->message_bytes_written < header_len)
630  {
631  bytes_written =
632  _dbus_write_socket_two (socket_transport->fd,
633  header,
634  socket_transport->message_bytes_written,
635  header_len - socket_transport->message_bytes_written,
636  body,
637  0, body_len);
638  }
639  else
640  {
641  bytes_written =
642  _dbus_write_socket (socket_transport->fd,
643  body,
644  (socket_transport->message_bytes_written - header_len),
645  body_len -
646  (socket_transport->message_bytes_written - header_len));
647  }
648 
649  saved_errno = _dbus_save_socket_errno ();
650  }
651  }
652 
653  if (bytes_written < 0)
654  {
655  /* EINTR already handled for us */
656 
657  /* If the other end closed the socket with close() or shutdown(), we
658  * receive EPIPE here but we must not close the socket yet: there
659  * might still be some data to read. See:
660  * http://lists.freedesktop.org/archives/dbus/2008-March/009526.html
661  */
662 
663  if (_dbus_get_is_errno_eagain_or_ewouldblock (saved_errno) || _dbus_get_is_errno_epipe (saved_errno))
664  goto out;
665 
666  /* Since Linux commit 25888e (from 2.6.37-rc4, Nov 2010), sendmsg()
667  * on Unix sockets returns -1 errno=ETOOMANYREFS when the passfd
668  * mechanism (SCM_RIGHTS) is used recursively with a recursion level
669  * of maximum 4. The kernel does not have an API to check whether
670  * the passed fds can be forwarded and it can change asynchronously.
671  * See:
672  * https://bugs.freedesktop.org/show_bug.cgi?id=80163
673  */
674 
675  else if (_dbus_get_is_errno_etoomanyrefs (saved_errno))
676  {
677  /* We only send fds in the first byte of the message.
678  * ETOOMANYREFS cannot happen after.
679  */
680  _dbus_assert (socket_transport->message_bytes_written == 0);
681 
682  _dbus_verbose (" discard message of %d bytes due to ETOOMANYREFS\n",
683  total_bytes_to_write);
684 
685  socket_transport->message_bytes_written = 0;
686  _dbus_string_set_length (&socket_transport->encoded_outgoing, 0);
687  _dbus_string_compact (&socket_transport->encoded_outgoing, 2048);
688 
689  /* The message was not actually sent but it needs to be removed
690  * from the outgoing queue
691  */
692  _dbus_connection_message_sent_unlocked (transport->connection,
693  message);
694  }
695  else
696  {
697  _dbus_verbose ("Error writing to remote app: %s\n",
698  _dbus_strerror (saved_errno));
699  do_io_error (transport);
700  goto out;
701  }
702  }
703  else
704  {
705  _dbus_verbose (" wrote %d bytes of %d\n", bytes_written,
706  total_bytes_to_write);
707 
708  total += bytes_written;
709  socket_transport->message_bytes_written += bytes_written;
710 
711  _dbus_assert (socket_transport->message_bytes_written <=
712  total_bytes_to_write);
713 
714  if (socket_transport->message_bytes_written == total_bytes_to_write)
715  {
716  socket_transport->message_bytes_written = 0;
717  _dbus_string_set_length (&socket_transport->encoded_outgoing, 0);
718  _dbus_string_compact (&socket_transport->encoded_outgoing, 2048);
719 
720  _dbus_connection_message_sent_unlocked (transport->connection,
721  message);
722  }
723  }
724  }
725 
726  out:
727  if (oom)
728  return FALSE;
729  else
730  return TRUE;
731 }

734 static dbus_bool_t
735 do_reading (DBusTransport *transport)
736 {
737  DBusTransportSocket *socket_transport = (DBusTransportSocket*) transport;
738  DBusString *buffer;
739  int bytes_read;
740  int total;
741  dbus_bool_t oom;
742  int saved_errno;
743 
744  _dbus_verbose ("fd = %d\n",socket_transport->fd);
745 
746  /* No messages without authentication! */
747  if (!_dbus_transport_try_to_authenticate (transport))
748  return TRUE;
749 
750  oom = FALSE;
751 
752  total = 0;
753 
754  again:
755 
756  /* See if we've exceeded max messages and need to disable reading */
757  check_read_watch (transport);
758 
759  if (total > socket_transport->max_bytes_read_per_iteration)
760  {
761  _dbus_verbose ("%d bytes exceeds %d bytes read per iteration, returning\n",
762  total, socket_transport->max_bytes_read_per_iteration);
763  goto out;
764  }
765 
766  _dbus_assert (socket_transport->read_watch != NULL ||
767  transport->disconnected);
768 
769  if (transport->disconnected)
770  goto out;
771 
772  if (!dbus_watch_get_enabled (socket_transport->read_watch))
773  return TRUE;
774 
775  if (_dbus_auth_needs_decoding (transport->auth))
776  {
777  /* Does fd passing even make sense with encoded data? */
778  _dbus_assert(!DBUS_TRANSPORT_CAN_SEND_UNIX_FD(transport));
779 
780  if (_dbus_string_get_length (&socket_transport->encoded_incoming) > 0)
781  bytes_read = _dbus_string_get_length (&socket_transport->encoded_incoming);
782  else
783  bytes_read = _dbus_read_socket (socket_transport->fd,
784  &socket_transport->encoded_incoming,
785  socket_transport->max_bytes_read_per_iteration);
786 
787  saved_errno = _dbus_save_socket_errno ();
788 
789  _dbus_assert (_dbus_string_get_length (&socket_transport->encoded_incoming) ==
790  bytes_read);
791 
792  if (bytes_read > 0)
793  {
794  _dbus_message_loader_get_buffer (transport->loader,
795  &buffer);
796 
797  if (!_dbus_auth_decode_data (transport->auth,
798  &socket_transport->encoded_incoming,
799  buffer))
800  {
801  _dbus_verbose ("Out of memory decoding incoming data\n");
802  _dbus_message_loader_return_buffer (transport->loader,
803  buffer);
804 
805  oom = TRUE;
806  goto out;
807  }
808 
809  _dbus_message_loader_return_buffer (transport->loader,
810  buffer);
811 
812  _dbus_string_set_length (&socket_transport->encoded_incoming, 0);
813  _dbus_string_compact (&socket_transport->encoded_incoming, 2048);
814  }
815  }
816  else
817  {
818  _dbus_message_loader_get_buffer (transport->loader,
819  &buffer);
820 
821 #ifdef HAVE_UNIX_FD_PASSING
822  if (DBUS_TRANSPORT_CAN_SEND_UNIX_FD(transport))
823  {
824  int *fds, n_fds;
825 
826  if (!_dbus_message_loader_get_unix_fds(transport->loader, &fds, &n_fds))
827  {
828  _dbus_verbose ("Out of memory reading file descriptors\n");
829  _dbus_message_loader_return_buffer (transport->loader, buffer);
830  oom = TRUE;
831  goto out;
832  }
833 
834  bytes_read = _dbus_read_socket_with_unix_fds(socket_transport->fd,
835  buffer,
836  socket_transport->max_bytes_read_per_iteration,
837  fds, &n_fds);
838  saved_errno = _dbus_save_socket_errno ();
839 
840  if (bytes_read >= 0 && n_fds > 0)
841  _dbus_verbose("Read %i unix fds\n", n_fds);
842 
843  _dbus_message_loader_return_unix_fds(transport->loader, fds, bytes_read < 0 ? 0 : n_fds);
844  }
845  else
846 #endif
847  {
848  bytes_read = _dbus_read_socket (socket_transport->fd,
849  buffer, socket_transport->max_bytes_read_per_iteration);
850  saved_errno = _dbus_save_socket_errno ();
851  }
852 
853  _dbus_message_loader_return_buffer (transport->loader,
854  buffer);
855  }
856 
857  if (bytes_read < 0)
858  {
859  /* EINTR already handled for us */
860 
861  if (_dbus_get_is_errno_enomem (saved_errno))
862  {
863  _dbus_verbose ("Out of memory in read()/do_reading()\n");
864  oom = TRUE;
865  goto out;
866  }
867  else if (_dbus_get_is_errno_eagain_or_ewouldblock (saved_errno))
868  goto out;
869  else
870  {
871  _dbus_verbose ("Error reading from remote app: %s\n",
872  _dbus_strerror (saved_errno));
873  do_io_error (transport);
874  goto out;
875  }
876  }
877  else if (bytes_read == 0)
878  {
879  _dbus_verbose ("Disconnected from remote app\n");
880  do_io_error (transport);
881  goto out;
882  }
883  else
884  {
885  _dbus_verbose (" read %d bytes\n", bytes_read);
886 
887  total += bytes_read;
888 
889  if (!_dbus_transport_queue_messages (transport))
890  {
891  oom = TRUE;
892  _dbus_verbose (" out of memory when queueing messages we just read in the transport\n");
893  goto out;
894  }
895 
896  /* Try reading more data until we get EAGAIN and return, or
897  * exceed max bytes per iteration. If in blocking mode of
898  * course we'll block instead of returning.
899  */
900  goto again;
901  }
902 
903  out:
904  if (oom)
905  return FALSE;
906  else
907  return TRUE;
908 }

929 static dbus_bool_t
930 socket_handle_watch (DBusTransport *transport,
931  DBusWatch *watch,
932  unsigned int flags)
933 {
934  DBusTransportSocket *socket_transport = (DBusTransportSocket*) transport;
935 
936  _dbus_assert (watch == socket_transport->read_watch ||
937  watch == socket_transport->write_watch);
938  _dbus_assert (watch != NULL);
939 
940  /* If we hit an error here on a write watch, don't disconnect the transport yet because data can
941  * still be in the buffer and do_reading may need several iteration to read
942  * it all (because of its max_bytes_read_per_iteration limit).
943  */
944  if (!(flags & DBUS_WATCH_READABLE) && unix_error_with_read_to_come (transport, watch, flags))
945  {
946  _dbus_verbose ("Hang up or error on watch\n");
947  _dbus_transport_disconnect (transport);
948  return TRUE;
949  }
950 
951  if (watch == socket_transport->read_watch &&
952  (flags & DBUS_WATCH_READABLE))
953  {
954  dbus_bool_t auth_finished;
955 #if 1
956  _dbus_verbose ("handling read watch %p flags = %x\n",
957  watch, flags);
958 #endif
959  if (!do_authentication (transport, TRUE, FALSE, &auth_finished))
960  return FALSE;
961 
962  /* We don't want to do a read immediately following
963  * a successful authentication. This is so we
964  * have a chance to propagate the authentication
965  * state further up. Specifically, we need to
966  * process any pending data from the auth object.
967  */
968  if (!auth_finished)
969  {
970  if (!do_reading (transport))
971  {
972  _dbus_verbose ("no memory to read\n");
973  return FALSE;
974  }
975  }
976  else
977  {
978  _dbus_verbose ("Not reading anything since we just completed the authentication\n");
979  }
980  }
981  else if (watch == socket_transport->write_watch &&
982  (flags & DBUS_WATCH_WRITABLE))
983  {
984 #if 1
985  _dbus_verbose ("handling write watch, have_outgoing_messages = %d\n",
986  _dbus_connection_has_messages_to_send_unlocked (transport->connection));
987 #endif
988  if (!do_authentication (transport, FALSE, TRUE, NULL))
989  return FALSE;
990 
991  if (!do_writing (transport))
992  {
993  _dbus_verbose ("no memory to write\n");
994  return FALSE;
995  }
996 
997  /* See if we still need the write watch */
998  check_write_watch (transport);
999  }
1000 #ifdef DBUS_ENABLE_VERBOSE_MODE
1001  else
1002  {
1003  if (watch == socket_transport->read_watch)
1004  _dbus_verbose ("asked to handle read watch with non-read condition 0x%x\n",
1005  flags);
1006  else if (watch == socket_transport->write_watch)
1007  _dbus_verbose ("asked to handle write watch with non-write condition 0x%x\n",
1008  flags);
1009  else
1010  _dbus_verbose ("asked to handle watch %p on fd %d that we don't recognize\n",
1011  watch, dbus_watch_get_socket (watch));
1012  }
1013 #endif /* DBUS_ENABLE_VERBOSE_MODE */
1014 
1015  return TRUE;
1016 }

1069 static void
1070 socket_do_iteration (DBusTransport *transport,
1071  unsigned int flags,
1072  int timeout_milliseconds)
1073 {
1074  DBusTransportSocket *socket_transport = (DBusTransportSocket*) transport;
1075  DBusPollFD poll_fd;
1076  int poll_res;
1077  int poll_timeout;
1078 
1079  _dbus_verbose (" iteration flags = %s%s timeout = %d read_watch = %p write_watch = %p fd = %d\n",
1080  flags & DBUS_ITERATION_DO_READING ? "read" : "",
1081  flags & DBUS_ITERATION_DO_WRITING ? "write" : "",
1082  timeout_milliseconds,
1083  socket_transport->read_watch,
1084  socket_transport->write_watch,
1085  socket_transport->fd);
1086 
1087  /* the passed in DO_READING/DO_WRITING flags indicate whether to
1088  * read/write messages, but regardless of those we may need to block
1089  * for reading/writing to do auth. But if we do reading for auth,
1090  * we don't want to read any messages yet if not given DO_READING.
1091  */
1092 
1093  poll_fd.fd = socket_transport->fd;
1094  poll_fd.events = 0;
1095 
1096  if (_dbus_transport_try_to_authenticate (transport))
1097  {
1098  /* This is kind of a hack; if we have stuff to write, then try
1099  * to avoid the poll. This is probably about a 5% speedup on an
1100  * echo client/server.
1101  *
1102  * If both reading and writing were requested, we want to avoid this
1103  * since it could have funky effects:
1104  * - both ends spinning waiting for the other one to read
1105  * data so they can finish writing
1106  * - prioritizing all writing ahead of reading
1107  */
1108  if ((flags & DBUS_ITERATION_DO_WRITING) &&
1109  !(flags & (DBUS_ITERATION_DO_READING | DBUS_ITERATION_BLOCK)) &&
1110  !transport->disconnected &&
1111  _dbus_connection_has_messages_to_send_unlocked (transport->connection))
1112  {
1113  do_writing (transport);
1114 
1115  if (transport->disconnected ||
1116  !_dbus_connection_has_messages_to_send_unlocked (transport->connection))
1117  goto out;
1118  }
1119 
1120  /* If we get here, we decided to do the poll() after all */
1121  _dbus_assert (socket_transport->read_watch);
1122  if (flags & DBUS_ITERATION_DO_READING)
1123  poll_fd.events |= _DBUS_POLLIN;
1124 
1125  _dbus_assert (socket_transport->write_watch);
1126  if (flags & DBUS_ITERATION_DO_WRITING)
1127  poll_fd.events |= _DBUS_POLLOUT;
1128  }
1129  else
1130  {
1131  DBusAuthState auth_state;
1132 
1133  auth_state = _dbus_auth_do_work (transport->auth);
1134 
1135  if (transport->receive_credentials_pending ||
1136  auth_state == DBUS_AUTH_STATE_WAITING_FOR_INPUT)
1137  poll_fd.events |= _DBUS_POLLIN;
1138 
1139  if (transport->send_credentials_pending ||
1140  auth_state == DBUS_AUTH_STATE_HAVE_BYTES_TO_SEND)
1141  poll_fd.events |= _DBUS_POLLOUT;
1142  }
1143 
1144  if (poll_fd.events)
1145  {
1146  int saved_errno;
1147 
1148  if (flags & DBUS_ITERATION_BLOCK)
1149  poll_timeout = timeout_milliseconds;
1150  else
1151  poll_timeout = 0;
1152 
1153  /* For blocking selects we drop the connection lock here
1154  * to avoid blocking out connection access during a potentially
1155  * indefinite blocking call. The io path is still protected
1156  * by the io_path_cond condvar, so we won't reenter this.
1157  */
1158  if (flags & DBUS_ITERATION_BLOCK)
1159  {
1160  _dbus_verbose ("unlock pre poll\n");
1161  _dbus_connection_unlock (transport->connection);
1162  }
1163 
1164  again:
1165  poll_res = _dbus_poll (&poll_fd, 1, poll_timeout);
1166  saved_errno = _dbus_save_socket_errno ();
1167 
1168  if (poll_res < 0 && _dbus_get_is_errno_eintr (saved_errno))
1169  goto again;
1170 
1171  if (flags & DBUS_ITERATION_BLOCK)
1172  {
1173  _dbus_verbose ("lock post poll\n");
1174  _dbus_connection_lock (transport->connection);
1175  }
1176 
1177  if (poll_res >= 0)
1178  {
1179  if (poll_res == 0)
1180  poll_fd.revents = 0; /* some concern that posix does not guarantee this;
1181  * valgrind flags it as an error. though it probably
1182  * is guaranteed on linux at least.
1183  */
1184 
1185  if (poll_fd.revents & _DBUS_POLLERR)
1186  do_io_error (transport);
1187  else
1188  {
1189  dbus_bool_t need_read = (poll_fd.revents & _DBUS_POLLIN) > 0;
1190  dbus_bool_t need_write = (poll_fd.revents & _DBUS_POLLOUT) > 0;
1191  dbus_bool_t authentication_completed;
1192 
1193  _dbus_verbose ("in iteration, need_read=%d need_write=%d\n",
1194  need_read, need_write);
1195  do_authentication (transport, need_read, need_write,
1196  &authentication_completed);
1197 
1198  /* See comment in socket_handle_watch. */
1199  if (authentication_completed)
1200  goto out;
1201 
1202  if (need_read && (flags & DBUS_ITERATION_DO_READING))
1203  do_reading (transport);
1204  if (need_write && (flags & DBUS_ITERATION_DO_WRITING))
1205  do_writing (transport);
1206  }
1207  }
1208  else
1209  {
1210  _dbus_verbose ("Error from _dbus_poll(): %s\n",
1211  _dbus_strerror (saved_errno));
1212  }
1213  }
1214 
1215 
1216  out:
1217  /* We need to install the write watch only if we did not
1218  * successfully write everything. Note we need to be careful that we
1219  * don't call check_write_watch *before* do_writing, since it's
1220  * inefficient to add the write watch, and we can avoid it most of
1221  * the time since we can write immediately.
1222  *
1223  * However, we MUST always call check_write_watch(); DBusConnection code
1224  * relies on the fact that running an iteration will notice that
1225  * messages are pending.
1226  */
1227  check_write_watch (transport);
1228 
1229  _dbus_verbose (" ... leaving do_iteration()\n");
1230 }
